Google Drive sync
Syncs all jobs, customers, and equipment across all your devices via Google Sheets.
Setup guide
1
Create a Google Sheet
Go to sheets.google.com and create a new blank sheet named PER Field Tracker Data.
2
Open Apps Script
In the sheet: Extensions → Apps Script. Delete all existing code.
3
Paste this script
function doGet(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var val = sheet.getRange("A1").getValue();
    var cb = e.parameter.callback;
    var json = JSON.stringify({data: val});
    if (cb) {
      return ContentService
        .createTextOutput(cb + "(" + json + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return ContentService
      .createTextOutput(json)
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var body = e.postData ? JSON.parse(e.postData.contents) : JSON.parse(e.parameter.payload);
    if (body.action === "write") {
      sheet.getRange("A1").setValue(body.data);
      return ContentService
        .createTextOutput(JSON.stringify({ok: true}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    return ContentService
      .createTextOutput(JSON.stringify({error: "Unknown"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
4
Deploy as Web App
Deploy → New deployment
Type: Web app
Execute as: Me
Access: Anyone
Copy the Web App URL and paste it above.
5
Install on home screen
In Safari: tap the Share button → Add to Home Screen. The app icon will appear on your home screen.
