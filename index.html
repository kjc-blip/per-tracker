1
Create a Google Sheet
Go to sheets.google.com and create a blank sheet named PER Field Tracker Data.
2
Open Apps Script
Extensions → Apps Script. Delete all existing code.
3
Paste this script
function doGet(e){try{var s=SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();var v=s.getRange("A1").getValue();var cb=e.parameter.callback;var j=JSON.stringify({data:v});if(cb){return ContentService.createTextOutput(cb+"("+j+")").setMimeType(ContentService.MimeType.JAVASCRIPT);}return ContentService.createTextOutput(j).setMimeType(ContentService.MimeType.JSON);}catch(err){return ContentService.createTextOutput(JSON.stringify({error:err.toString()})).setMimeType(ContentService.MimeType.JSON);}}
function doPost(e){try{var s=SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();var b=e.postData?JSON.parse(e.postData.contents):JSON.parse(e.parameter.payload);if(b.action==="write"){s.getRange("A1").setValue(b.data);return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);}}catch(err){return ContentService.createTextOutput(JSON.stringify({error:err.toString()})).setMimeType(ContentService.MimeType.JSON);}}
4
Deploy as Web App
Deploy → New deployment → Web app → Execute as: Me → Access: Anyone. Copy URL and paste above.
5
Install on iPhone
Open GitHub Pages URL in Safari → Share button → Add to Home Screen.
